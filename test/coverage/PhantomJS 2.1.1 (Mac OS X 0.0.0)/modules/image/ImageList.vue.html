 * 图片列表组件<br/>

 *            <sw-image  :src="url"
 *                       :defaultSrc="defaultSrc"
 *                       lazy></sw-image>

 *        <dd>
 *            <p class="ph-image-title">图片</p>

 *        </dd>
 *    </dl>
 *</sw-image-list>
 * ```
 *
 * @class ImageList
 * @module 操作类组件

 * @Constructor

 * */

    name: 'SwImageList',
    extends: SwBase,
    props: {

            type: String,
            default: 'image-list'

        /**
         * 图片列数
         * @property column
         * @type Number
         * @default 1
         * */
        column: {
            type: Number,

}
