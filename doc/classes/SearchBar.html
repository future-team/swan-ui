    SearchBar

    SearchBar

    • 可通过buttonText设置按钮的文字

    属性：

    • buttonText: 按钮文字
    • placeholder: 输入提示

    事件：

    • input／change: 回调参数(value,evt)，输入时触发
    • focus: 聚焦时触发
    • blur: 失去焦点时触发
    • search: 回调参数(value,evt)，按下回车时触发
    • click: 回调参数(value,evt)，按钮被点击时触发

    <script>
    const demos = [
        {
            title: '基本用法',
            desc: '',
            code: '<sw-search></sw-search>'
        },
        {
            title: '定制按钮',
            desc: 'buttonText',
            code: '<sw-search buttonText="确定"></sw-search>'
        },
        {
            title: 'click事件',
            desc: '点击取消，触发click，另外还有blur、focus事件',
            code: `
                <sw-search v-model="searchValue"
                           @click="handleClick"
                           buttonText="取消">
                </sw-search>
            `,
            options: {
                data(){
                    return {
                        searchValue: ''
                    }
                },
                methods: {
                    handleClick(){
                        this.searchValue = ''
                    }
                }
            }
        },
        {
            title: 'search事件',
            desc: '点击回车，触发search',
            code: `
                <sw-search @search="handleSearch"
                           @click="handleSearch"
                           buttonText="确定" />
            `,
            options: {
                methods: {
                    handleSearch(val){
                        alert(点击回车或确定，搜索：${val})
                    }
                }
            }
        }
    ]
    import Base from '../base.vue'
    export default {
        name: 'PageSearch',
        extends:Base,
        data(){
            return {
                demos: demos
            }
        }
    }
</script>
